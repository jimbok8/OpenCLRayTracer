430 inline bool planeBoxOverlap(CL_FLOAT3 normal, CL_FLOAT3 vert, CL_FLOAT3 maxbox)
431 {
432  CL_FLOAT3 vmin,vmax;
433 
434  //Checking X
435  vmin.x=-sign(normal.x) * maxbox.x - vert.x;
436  vmax.x= sign(normal.x) * maxbox.x - vert.x;
437

460 inline bool AABBTriangleIntersect(CL_FLOAT3 boxcenter,CL_FLOAT3 boxhalfsize, CL_FLOAT3 v0,CL_FLOAT3 v1, CL_FLOAT3 v2)
461 {
462 
463 
464  /* use separating axis theorem to test overlap between triangle and box */
465  /* need to test for overlap in these directions: */
466  /* 1) the {x,y,z}-directions (actually, since we use the AABB of the triangle */
467  /* we do not even need to test these) */
468  /* 2) normal of the triangle */
469  /* 3) crossproduct(edge from tri, {x,y,z}-directin) */
470  /* this gives 3x3=9 more tests */
471 
472  float minimum,maximum;
473  bool yesNoFlag = true;
474 
475  /* move everything so that the boxcenter is in (0,0,0) */
476 
477  v0 = boxcenter - v0;
478  v1 = boxcenter - v1;
479  v2 = boxcenter - v2;
480 
481  /* Bullet 3: */
482  /* test the 9 tests first (this was faster) */
483  {
484  CL_FLOAT3 edge = v1 - v0;
485  float p0,p1,p2,rad,fex,fey,fez;
486 
487  fex = fabs(edge.x);
488  fey = fabs(edge.y);
489  fez = fabs(edge.z);
490  AXISTEST_X01(edge.z, edge.y, fez, fey);
491  AXISTEST_Y02(edge.z, edge.x, fez, fex);
492  AXISTEST_Z12(edge.y, edge.x, fey, fex);
493 
494  edge = v2 - v1;
495  fex = fabs(edge.x);
496  fey = fabs(edge.y);
497  fez = fabs(edge.z);
498  AXISTEST_X01(edge.z, edge.y, fez, fey);
499  AXISTEST_Y02(edge.z, edge.x, fez, fex);
500  AXISTEST_Z0(edge.y, edge.x, fey, fex);
501 
502  edge = v0 - v2;
503  fex = fabs(edge.x);
504  fey = fabs(edge.y);
505  fez = fabs(edge.z);
506 
507  AXISTEST_X2(edge.z, edge.y, fez, fey);
508  AXISTEST_Y1(edge.z, edge.x, fez, fex);
509  AXISTEST_Z12(edge.y, edge.x, fey, fex);
510  }
511 
512  /* Bullet 1: */
513  /* first test overlap in the {x,y,z}-directions */
514  /* find min, max of the triangle each direction, and test for overlap in */
515  /* that direction -- this is equivalent to testing a minimal AABB around */
516  /* the triangle against the AABB */
517 
518  /* test in X-direction */
519 
520  minimum = min(v0.x,min(v1.x,v2.x));
521  maximum = max(v0.x,max(v1.x,v2.x));
522  yesNoFlag = yesNoFlag && (minimum <= boxhalfsize.x && maximum >= -boxhalfsize.x);
523 
524  /* test in Y-direction */
525  minimum = min(v0.y,min(v1.y,v2.y));
526  maximum = max(v0.y,max(v1.y,v2.y));
527  yesNoFlag = yesNoFlag && (minimum <= boxhalfsize.y && maximum >= -boxhalfsize.y);
528 
529  /* test in Z-direction */
530  minimum = min(v0.z,min(v1.z,v2.z));
531  maximum = max(v0.z,max(v1.z,v2.z));
532  yesNoFlag = yesNoFlag && (minimum <= boxhalfsize.z && maximum >= -boxhalfsize.z);
533 
534  /* Bullet 2: */
535  /* test if the box intersects the plane of the triangle */
536  /* compute plane equation of triangle: normal*x+d=0 */
537  CL_FLOAT3 normal = cross(v1 - v0,v2 - v1);
538  yesNoFlag = yesNoFlag && planeBoxOverlap(normal,v0,boxhalfsize);
539  return yesNoFlag;
540 }
