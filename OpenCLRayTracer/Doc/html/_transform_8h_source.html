88  inline void normalizeQuaternion(struct Quaternion* q)
89  {
90  CL_FLOAT d = q->r*q->r+q->i*q->i+q->j*q->j+q->k*q->k;
91 
92  // Check for zero length quaternion, and use the no-rotation
93  // quaternion in that case.
94  if (d < FLT_EPSILON)
95  {
96  q->i = q->j = q->k = 0.0f;
97  q->r = 1;
98  return;
99  }
100 
101  d = ((CL_FLOAT)1.0)/sqrt(d);
102  q->r *= d;
103  q->i *= d;
104  q->j *= d;
105  q->k *= d;
106  }

219 inline void fillRotate(struct Matrix4 *A, CL_FLOAT3 eulerAngles)
220 {
221  /* Fill the rotation matrix, using Euler angles */

236  x.data[0]=1; x.data[1]=0; x.data[2] =0; x.data[3] =0;
237  x.data[4]=0; x.data[5]=cx; x.data[6] =-sx; x.data[7] =0;
238  x.data[8]=0; x.data[9]=sx; x.data[10]=cx; x.data[11]=0;
239 
240  y.data[0]=cy; y.data[1]=0; y.data[2] =sy; y.data[3] =0;
241  y.data[4]=0; y.data[5]=1; y.data[6] =0; y.data[7] =0;
242  y.data[8]=-sy; y.data[9]=0; y.data[10]=cy; y.data[11]=0;
243 
244  z.data[0]=cz; z.data[1]=-sz; z.data[2] =0; z.data[3] =0;
245  z.data[4]=sz; z.data[5]=cz; z.data[6] =0; z.data[7] =0;
246  z.data[8]=0; z.data[9]=0; z.data[10]=1; z.data[11]=0;
247 
248  /* Note we are multiplying x*y*z. You can change the order,
249  but you will get different results. */

251  matrixMultiply(&z,&y,&temp); // multiply 2 matrices
252  matrixMultiply(&temp,&x,A); // multiply result by 3rd matrix
