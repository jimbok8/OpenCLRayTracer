359  inline Common::Result getMaxWorkItemSizes(boost::shared_array<size_t>& value, Common::Errata& err) const
360  {
361  char buf[1024];
362  cl_int status = clGetDeviceInfo(_deviceId,CL_DEVICE_MAX_WORK_ITEM_SIZES,sizeof(buf),buf,NULL);
363  if (status != CL_SUCCESS)
364  {
365  FILL_ERRATA(err,"Getting device property: "<< std::string("CL_DEVICE_MAX_WORK_ITEM_SIZES") <<" failed!, reason: " << appsdk::getOpenCLErrorCodeStr(status));
366  return Common::Error;
367  }
368  cl_uint dimensions = 0;
369  if (!getMaxWorkItemDimensions(dimensions,err))
370  return Common::Error;
371  value.reset(new size_t[dimensions]);
372  size_t* output = (size_t*)buf;
373  for (cl_uint i = 0; i < dimensions; i++)
374  {
375  value[i] = *output;
376  output++;
377  }
378  return Common::Success;
379  }
